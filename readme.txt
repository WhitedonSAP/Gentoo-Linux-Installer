---> root # mirrorselect -i -o >> /etc/portage/make.conf

---------------------------------------------------------------------

2. Add permission to exec the installer:

---> root # chmod +x Gentoo-Linux-Installer.sh

---------------------------------------------------------------------

3. Run the installer:

---> root # ./Gentoo-Linux-Installer.sh

---> root # sh Gentoo-Linux-Installer.sh
